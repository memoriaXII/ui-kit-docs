## Usage

`forwardRef` exposes Arco's `SearchBarRef` for imperative focus / blur.

```tsx
import { SearchBar } from "@appboxo/ui-kit"
import { useRef } from "react"
import type { SearchBarRef } from "@arco-design/mobile-react/cjs/search-bar"

export function Example() {
  const ref = useRef<SearchBarRef>(null)

  return (
    <SearchBar
      ref={ref}
      placeholder="Search rewards, stores, brands…"
      onChange={(value) => console.log(value)}
    />
  )
}
```

## Theme tokens

`--search-bar-input-wrapper-background-color`, `--search-bar-*` (all required).
