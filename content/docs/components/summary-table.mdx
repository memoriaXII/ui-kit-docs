## Usage

```tsx
import { SummaryTable } from "@appboxo/ui-kit"

export function Example() {
  return (
    <SummaryTable
      rows={[
        { label: "Subtotal", value: "$120.00" },
        { label: "Tax", value: "$9.60" },
        { label: "Total", value: "$129.60", emphasised: true },
      ]}
    />
  )
}
```

Set `emphasised: true` on the row that should stand out (typically the total).
