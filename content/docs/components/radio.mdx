## Usage

```tsx
import { RadioGroup, RadioGroupItem, Label } from "@freedom/ui-kit"

export function Example() {
  return (
    <RadioGroup defaultValue="card">
      <RadioGroupItem value="card" id="card" />
      <Label htmlFor="card">Credit card</Label>
    </RadioGroup>
  )
}
```
