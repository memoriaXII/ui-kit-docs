## Usage

Exported as `TextArea` (not `Textarea`). Same `hasError` flag as `Input`.

```tsx
import { TextArea } from "@appboxo/ui-kit"

export function Example() {
  return (
    <TextArea
      placeholder="Tell us what happened"
      rows={4}
      autoSize={{ minRows: 4, maxRows: 8 }}
    />
  )
}
```
